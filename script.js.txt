function initMap() {
    var propiedad = {lat: 19.432608, lng: -99.133209}; // Coordenadas de ejemplo
    var map = new google.maps.Map(document.getElementById('map'), {
        zoom: 15,
        center: propiedad
    });

    var marker = new google.maps.Marker({
        position: propiedad,
        map: map
    });
}
